// improvement : both async task executing parallel and wait for both to completing.
                    async function(){
                        const d1Promise = axios.get('//api url')
                        const d2Promise = axios.get('//api url')

                        const [d1, d2] = await Promise.all([d1Promise, d2Promise])
                        const result = `
                            <h1>${d1.name}</h1>
                            <h2>${d2.name}</h2>
                        `
                    }

// create HOF (High order function) :
                    function handleError (fn) {
                        return function (...params) {
                            return fn(...params).catch(function (err) {
                                //do something with error
                                console.log('Oops! ,', err)
                            })
                        }
                    }
//shorthand
                    const handleError = fn => (...params) => fn(...params).catch(console.error)
